setup-software-gl

Enables software GL rendering using the Mesa llvmpipe driver. Persists the environment variables in ~/.bashrc and optionally disables X11 compositing for XFWM4.

function --no-persist --no-composite
View Code 
setup_software_gl() {
  local persist=1 composite=1
  while [[ "${1:-}" == --* ]]; do
    case "$1" in
      --no-persist)   persist=0; shift ;;
      --no-composite) composite=0; shift ;;
      --help|-h) echo -e "📖 \033[1msetup-software-gl\033[0m [--no-persist] [--no-composite]"; return 0 ;;
      *) echo -e "❌ Unknown flag: $1"; return 1 ;;
    esac
  done
  local vars='export LIBGL_ALWAYS_SOFTWARE=1
export GALLIUM_DRIVER=llvmpipe
export MESA_GL_VERSION_OVERRIDE=3.3'
  if (( persist )); then
    if ! grep -q "LIBGL_ALWAYS_SOFTWARE" ~/.bashrc; then
      printf '\n%s\n' "$vars" >> ~/.bashrc
    fi
  fi
  export LIBGL_ALWAYS_SOFTWARE=1
  export GALLIUM_DRIVER=llvmpipe
  export MESA_GL_VERSION_OVERRIDE=3.3
}
alias setup-software-gl='setup_software_gl'

show-greeter

Detects the active display manager and shows its greeter configuration . Supports LightDM , GDM3 , SDDM , LXDM , XDM and SLiM .

function
View Code 
show_greeter() {
  local dm
  dm=$(cat /etc/X11/default-display-manager 2>/dev/null | xargs basename)
  if [[ -z "$dm" ]]; then
    dm=$(systemctl status display-manager 2>/dev/null | grep -oP '(?<=Loaded: loaded \().*?(?=;)' | xargs basename)
  fi
  case "$dm" in
    lightdm)
      grep -r "greeter-session" /etc/lightdm/ 2>/dev/null | grep -v "^#"
      ;;
    gdm3|gdm)
      echo "=== GDM3 Display Manager ==="
      cat /etc/gdm3/daemon.conf 2>/dev/null
      cat /etc/gdm3/custom.conf 2>/dev/null
      ;;
    sddm)
      grep -i "theme\|greeter" /etc/sddm.conf /etc/sddm.conf.d/*.conf 2>/dev/null
      ;;
    *)
      echo "Unknown DM: '$dm'"
      ;;
  esac
}
alias show-greeter='show_greeter'

Network Procedures

Network probing, bandwidth monitoring, and power consumption tracking utilities.

net-probe

Probes a network host with ARP , ping , netcat and route query. Full network diagnostic in a single command.

function $1: IP $2: port --gateway --local
View Code 
net_probe() {
  local ip="" port=""
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --gateway)
        ip=$(ip route | awk '/default/{print $3; exit}')
        shift ;;
      --local)
        ip="192.168.1.1"; port="22"
        shift ;;
      *)
        if [ -z "$ip" ]; then ip="$1"; else port="$1"; fi; shift ;;
    esac
  done
  arp -a 2>/dev/null
  ping -c 4 "$ip"
  [ -n "$port" ] && nc -vz "$ip" "$port"
  ip route get "$ip"
}
alias net-probe='net_probe'

cat-band

Measures bandwidth usage between start and Enter. Based on the script by Luke Smith . Reads from /sys/class/net/[ew]*/statistics/rx_bytes .

function
View Code 
cat_band() {
  init="$(($(cat /sys/class/net/[ew]*/statistics/rx_bytes | paste -sd '+')))"
  printf "Recording bandwidth.. Press enter to stop."
  read -r "stop"
  fin="$(($(cat /sys/class/net/[ew]*/statistics/rx_bytes | paste -sd '+')))"
  printf "%4sB of bandwidth used.\\n" $(numfmt --to=iec $(($fin-$init)))
}
alias cat-band='cat_band'

track-power-usage

Runs powerstat (RAPL) and saves the output to a timestamped log in ~/.logs/ .

function $1: duration (3600) $2: tick (1)
View Code 
track_power_usage() {
  local duration=${1:-3600}
  local tick=${2:-1}
  if ! command -v powerstat &>/dev/null; then
    echo -e "❌ \033[1;31mpowerstat not found.\033[0m Install with: sudo apt install powerstat"
    return 1
  fi
  mkdir -p "$HOME/.logs"
  local log="$HOME/.logs/powerstat-$(date +%Y-%m-%d_%H-%M-%S).log"
  echo -e "⚡ \033[1;36mRecording power stats for ${duration}s (tick: ${tick}s) → $log\033[0m"
  sudo stdbuf -oL powerstat -R "$tick" "$duration" | tee "$log"
}
alias track-power-usage='track_power_usage'

wc-l-novendors

Counts total lines in the directory, excluding vendor/ , node_modules/ and .git/ . Displays the running total during execution.

function
View Code 
wc_lines_novendors() {
  local files=0 total=0
  while IFS= read -r -d $'\0' file; do
    lines=$(wc -l < "$file")
    total=$((total+lines))
    files=$((files+1))
    printf "%s %d\n" "$file" "$total"
  done < <(find . -type f \
        ! -path '*/vendor/*' \
        ! -path '*/node_modules/*' \
        ! -path '*/.git/*' \
        -print0)
  echo "-> TOTAL: $total lines in $files files"
}
alias wc-l-novendors='wc_lines_novendors'

hfold

Shows the heaviest folders by size in GB.

function $1: root (.) $2: top N (10)
View Code 
heavy_folders() {
  local root=${1:-.} top=${2:-10}
  du -Sb "$root"/* 2>/dev/null \
    | sort -nr \
    | head -n "$top" \
    | awk '{printf "%.2fGB\t%s\n",$1/1024/1024/1024,$2}'
}
alias hfold='heavy_folders'

  sudo cat /etc/os-release 2>/dev/null | sed \
    -e "s/^\([A-Z_]*\)=/\033[1;34m\1\033[0m=/" | _pretty_nl
